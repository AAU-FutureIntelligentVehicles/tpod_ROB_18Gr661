1function [theta0,xout,yout,alpha,carVec] = fcn(x0,y0,theta0,PointMatrix)
 2
 3% Define constants
 4LOWERDIST = 1;
 5UPPERDIST = 1.3;
 6VIEWING_ANGLE = 130*pi()/180;  %130 degrees in radians
 7WHEEL_BASE = 1.62;
 8
 9% find the path length
10szpm = size(PointMatrix);
11
12% initialize the variables. 
13xout = PointMatrix(szpm(1,1),1);
14yout = PointMatrix(szpm(1,1),2);
15alpha = 0;
16carVec = [cos(theta0); sin(theta0)];
17
18for i = 1:szpm(1,1)
19    %translate to the front of the cart.
20    car_x = x0 + cos(theta0) * WHEEL_BASE;
21    car_y = y0 + sin(theta0) * WHEEL_BASE;
22    
23    distance = sqrt((PointMatrix(i,1)-car_x)^2+(PointMatrix(i,2)-car_y)^2);
24
25    if distance >= LOWERDIST && distance <= UPPERDIST 
26        % creates a vector from the car front point to the pathpoint
27        % substract our glabal position and the translation to the front. 
28        pointVec = [PointMatrix(i,1)-x0-cos(theta0)*WHEEL_BASE ; PointMatrix(i,2)-y0-sin(theta0)*WHEEL_BASE];
29        % creates a vector along the car
30        carVec = [cos(theta0); sin(theta0)];
31        
32        % find the angle using the formular for finding angle between two
33        % vectors.
34        alpha = acos(dot(carVec,pointVec)/(norm(carVec)*norm(pointVec)));
35
36        if alpha <= VIEWING_ANGLE/2
37            xout = PointMatrix(i,1);
38            yout = PointMatrix(i,2);
39            %find the direction of the angle using the cross product.
40            alpha = alpha * sign(carVec(1,1)*pointVec(2,1)-carVec(2,1)*pointVec(1,1));
41            break %break the loop as soon as we find a point 
42        end          
43    end     
44end
